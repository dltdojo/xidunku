prerequisites

  • java

install mdbook-kubook

$ cargo install mdbook-kubook

book.toml

[preprocessor.kubook]
command = "mdbook-kubook"
renderer = ["html"]
java-path="/usr/bin/java"
plantuml-jar="/usr/local/share/plantuml.jar"
